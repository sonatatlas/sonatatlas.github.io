title: Limit the quantity of process

```
package main
import (
	"log"
	"time"
)

func main() {
	start := time.Now()
	ch := make(chan int, 2)
	for i := 0; i <= 10; i++ {
		ch <- 1
		go worker(i, ch)
	}
	close(ch)
	log.Println("complete", time.Since(start).Seconds())
}

func worker(i int, ch chan int) {
	log.Println("worker", i)
	time.Sleep(time.Second)
	<-ch
```
